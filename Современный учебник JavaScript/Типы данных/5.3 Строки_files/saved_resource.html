                                                                                                  Сделал функцию, которая разделяет строку на слова, записанные в объекте, будет очень удобно если нужно будет проверить каждое слово на грамматику.                                                                                                   let str = 'Widget with id';

                                                                                                  function getWords(string) {
                                                                                                  let words = {};
                                                                                                  let strLength = string.length;
                                                                                                  let spaces = 0;
                                                                                                  for (let i = 0; i < strLength; i++) {
                                                                                                  if (string[i] == " ") {
                                                                                                  ++spaces;
                                                                                                  } else {
                                                                                                  if (words[`Word${spaces}`] == undefined) {
                                                                                                  words[`Word${spaces}`] = string[i];
                                                                                                  } else {
                                                                                                  words[`Word${spaces}`] = words[`Word${spaces}`] + string[i];
                                                                                                  }
                                                                                                  }
                                                                                                  }
                                                                                                  return words;
                                                                                                  }

                                                                                                  let mywords = getWords(str);

                                                                                                  alert(`${mywords["Word0"]} ${mywords["Word1"]} ${mywords["Word2"]}`)

                                                                                                      let str = "Widget with id";

                                                                                                      function getWords( string ) {
                                                                                                      let words = [];
                                                                                                      let currentWord = '';

                                                                                                      for(let i = 0; i < string.length; i++) {
                                                                                                      let char = string[ i ];

                                                                                                                                        Больше символов - не значит читабельнее.

                                                                                                                                        Во-первых, крайне нечитабельные названия переменных. Не нужно экономить на символах. Если вы зачем-то хотите сохранить str.length(хотя вообще непонятно зачем это нужно, об этом ниже), то называйте переменную хотя бы str_length. А лучше используйте общепринятый camelCase - strLength.

                                                                                                                                        Во-вторых, не нужно использовать лишние переменные. В str_res вы записываете значение и сразу же его возвращаете. И вообще непонятно зачем оно объявлено до блока if
                                                                                                                                        Если вам очень нужно записать в отдельную переменную, то создавайте её в блоке, где её используете
                                                                                                                                        if (str_l > maxlength) {
                                                                                                                                        let str_res = str.slice(0, maxlength - str_l) + "…";
                                                                                                                                        return str_res;
                                                                                                                                        }
                                                                                                                                        Но лучше
                                                                                                                                        if (str_l > maxlength) {
                                                                                                                                        return str.slice(0, maxlength - str_l) + "…";
                                                                                                                                        }

                                                                                                                                        В-третьих, сохранять length в переменную не имеет никакого смысла в данном случае. Еинственной случай когда это имеет смысл - во время итерации. В данном же случае эта переменная только мешает и ухудшает читабельность.

                                                                                                                                        В-четвёртых, непонятно почему мы из maxLength вычитаем длину строки

                                                                                                                                        И в итоге приходим к такому коду:
                                                                                                                                        function truncate(str, maxLength) {
                                                                                                                                        if (str.length > maxLength) {
                                                                                                                                        return str.slice(0, maxLength + 1) + "...";
                                                                                                                                        }
                                                                                                                                        return str;
                                                                                                                                        }

                                                                                                                                        Этот код максимально читабельный, советую использовать такой стиль.
